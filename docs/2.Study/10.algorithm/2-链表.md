```java
ublic class SingleLinkedList {
    
    public static Node reverseList(Node head){
        Node pre=null;
        Node next=null;
        while(head!=null){
            next=head.next;
            head.next=pre;
            head=pre;
            head=next;
        }
        return pre;
    }
}

class Node{
           public int val;
           public Node next;
           public Node(int val) {
                   this.val = val;
            }
}
```
